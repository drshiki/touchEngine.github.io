Personal collection for C

This article is about some my study notes, and I write down these prevent forgetting when want to ues, and maybe this page is useful for your work or interview of your ideal work or study for C, or not. whatever, it is worth remaining some thing for for my study experience in C, and every topic description have a copy of compileable and executable code for any compiler meeting ANSI C99. and I will update this page continuedly.

exchange the two values of two variable without the temporary variable

				#include <stdio.h>
				int main()
				{
					int a=1,b=2;
					b=a^b;
					a=a^b;
					b=a^b;
					printf("a = %d, b = %d", a, b);//a = 2; b = 1
					return 0;
				}

get the file size through the standard funtions in a quickly way

			#include <stdio.h>
			int main()
			{
				FILE *fp = fopen("test.txt","r");
				/* test.txt contains the following content 
				 * hello world! this is a test file.
				*/
				
				if(fp){
					if(!fseek(fp, 0, SEEK_END))
						printf("%ld", ftell(fp)); //the output is file size
				}
				
				return 0;
			}

array's special repretentation

			#include <stdio.h>
			#include <string.h>
			int main()
			{
				for(int i = 0; i < strlen("mary");i++)
					printf("%c\t%c\n", "Mary"[i],i["Mary"]);
				/* the output is folow
				 * M    M
				 * a    a
				 * r    r
				 * y    y
				 */
				return 0;
			}

magical operator effect

			#include <stdio.h>
			int main()
			{
				int a = 10, b = 20;
				printf("a = %d\n",-~a); // -~a == a+=1;
				printf("b = %d\n",~-b); // ~-b == b-=1;
				return 0;
			}
